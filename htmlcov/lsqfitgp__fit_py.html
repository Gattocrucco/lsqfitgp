34@autograd.wrap_util.unary_to_nary 

35def jac(fun, x): 

36 """Like autograd.jacobian but with forward mode""" 

37 jvp = autograd.core.make_jvp(fun, x) 

38 ans = fun(x) 

39 vs = autograd.extend.vspace(x) 

40 grads = map(lambda b: jvp(b)[1], vs.standard_basis()) 

41 return np.reshape(np.stack(grads, axis=-1), ans.shape + vs.shape)
