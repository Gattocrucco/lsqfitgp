31def _hurwitz_zeta_series(m, x, a1, onlyeven=False, onlyodd=False, skipterm=None): 1feabcd

32 """ 

33 hurwitz zeta(s = m + x, a = 1 - a1) with integer m 

34 meant to be used with |x| ≤ 1/2, but no actual restriction 

35 assuming -S <= s <= 0 and |a1| <= 1/2 with S ~ some decade 

36 https://dlmf.nist.gov/25.11.E10 

37 """ 

38 

39 # decide number of terms to sum 

40 t = _patch_jax.float_type(m, x, a1) 1eabcd

41 nmax = _hze_nmax(t) 1eabcd

42 n = jnp.arange(nmax + 1) 1eabcd

43 

44 # make arguments broadcastable with n 

45 x = x[..., None] 1eabcd

46 m = m[..., None] 1eabcd

47 a1 = a1[..., None] 1eabcd

48 if skipterm is not None: 1eabcd

49 skipterm = skipterm[..., None] 1eabcd

50 

51 # compute pochhammer symbol, factorial and power terms 

52 nm = n + m 1eabcd

53 ns1 = nm - 1 + x # = n + s - 1 1eabcd

54 ns1_limit = jnp.where(ns1 == 0, 1, ns1) # pochhammer zero cancels zeta pole 1eabcd

55 ns1_limit = jnp.where(ns1 == 1, 0, ns1_limit) 1eabcd

56 # TODO this == 1 worries me, maybe sometimes it's violated, use shift 

57 factor = jnp.cumprod((ns1_limit * a1 / n).at[..., 0].set(1), -1, t) 1eabcd

58 

59 # handle tweaks to the series 

60 if onlyeven: 1eabcd

61 sl = slice(None, None, 2) 1eabcd

62 elif onlyodd: 1abcd

63 sl = slice(1, None, 2) 1abcd

64 if onlyeven or onlyodd: 1eabcd

65 n = n[sl] 1eabcd

66 nm = nm[..., sl] 1eabcd

67 ns1 = ns1[..., sl] 1eabcd

68 factor = factor[..., sl] 1eabcd

69 if skipterm is not None: 1eabcd

70 factor = jnp.where(n == skipterm, 0, factor) 1eabcd

71 

72 # compute zeta term 

73 zet = zeta(x, nm) # = zeta(n + s) 1eabcd

74 zet_limit = jnp.where(ns1 == 0, 1, zet) # pole cancelled by pochhammer 1eabcd

75 

76 # sum series 

77 kw = dict(precision=lax.Precision.HIGHEST) 1eabcd

78 series = jnp.matmul(factor[..., None, :], zet_limit[..., :, None], **kw) 1eabcd

79 return series.squeeze((-2, -1)) 1eabcd

86def hurwitz_zeta(s, a): 1feabcd

87 """ 

88 For 0 <= a <= 1 and -S <= s <= 0 with S not too large 

89 """ 

90 s = jnp.asarray(s) 1abcd

91 a = jnp.asarray(a) 1abcd

92 

93 cond = a < 1/2 # do a + 1 to bring a closer to 1 1abcd

94 a1 = jnp.where(cond, -a, 1. - a) 1abcd

95 zero = jnp.array(0) 1abcd

96 zeta = _hurwitz_zeta_series(zero, s, a1) # https://dlmf.nist.gov/25.11.E10 1abcd

97 zeta += jnp.where(cond, a ** -s, 0) # https://dlmf.nist.gov/25.11.E3 1abcd

98 return zeta 1abcd

104def periodic_zeta(x, s, imag=False): 1feabcd

105 """ 

106 compute F(x,s) = Li_s(e^2πix) for real s > 1, real x 

107 """ 

108 

109 x = jnp.asarray(x) 1eabcd

110 s = jnp.asarray(s) 1eabcd

111 

112 # decide boundary for large/small s implementation 

113 t = _patch_jax.float_type(x, s) 1eabcd

114 eps = jnp.finfo(t).eps 1eabcd

115 nmax = 50 1eabcd

116 larges = math.ceil(-math.log(eps) / math.log(nmax)) # 1/nmax^s < eps 1eabcd

117 

118 z_smalls = _periodic_zeta_smalls(x, s, imag) 1eabcd

119 z_larges = _periodic_zeta_larges(x, s, nmax, imag) 1eabcd

120 

121 return jnp.where(s < larges, z_smalls, z_larges) 1eabcd

141def _periodic_zeta_larges(x, s, nmax, imag): 1feabcd

142 """ https://dlmf.nist.gov/25.13.E1 """ 

143 

144 t = _patch_jax.float_type(x, s) 1eabcd

145 s = s.astype(t) # avoid n^s overflow with integer s 1eabcd

146 n = jnp.arange(1, nmax + 1) 1eabcd

147 neg, nx = _standard_x(n * x[..., None]) 1eabcd

148 func = jnp.sin if imag else jnp.cos 1eabcd

149 terms = func(2 * jnp.pi * nx) / n ** s[..., None] 1eabcd

150 if imag: 1eabcd

151 terms *= jnp.where(neg, -1, 1) 1abcd

152 return jnp.sum(terms, -1) 1eabcd

154def _periodic_zeta_smalls(x, s, imag): 1feabcd

155 """ 

156 https://dlmf.nist.gov/25.11.E10 and https://dlmf.nist.gov/25.11.E3 expanded 

157 into https://dlmf.nist.gov/25.13.E2 

158 """ 

159 neg, x = _standard_x(x) # x in [0, 1/2] 1eabcd

160 

161 eps = jnp.finfo(_patch_jax.float_type(x, s)).eps 1eabcd

162 s = jnp.where(s % 1, s, s * (1 + eps)) # avoid integer s 1eabcd

163 

164 s1 = 1 - s # < 0 1eabcd

165 q = -jnp.around(s1).astype(int) 1eabcd

166 a = s1 + q 1eabcd

167 # now s1 == -q + a with q integer >= 0 and |a| <= 1/2 

168 

169 pi = (2 * jnp.pi) ** -s1 1eabcd

170 gam = _gamma.gamma(s1) 1eabcd

171 func = _sin_pi2 if imag else _cos_pi2 1eabcd

172 pha = func(-q, a) # = sin or cos(π/2 s1), numerically accurate for small a 1eabcd

173 hzs = 2 * _hurwitz_zeta_series(-q, a, -x, onlyeven=not imag, onlyodd=imag, skipterm=q) 1eabcd

174 # hzs = ζ(s1,1+x) -/+ ζ(s1,1-x) but without the x^q term in the series 

175 pdiff = zeta_series_power_diff(x, q, a) 1eabcd

176 # pdiff accurately handles the sum of the external power x^-s1 due to 

177 # 25.11.E3 with the q-th term (cancellation) with even q for the real part 

178 # and odd q for the imaginary part 

179 cancelcond = jnp.logical_and(imag, q % 2 == 1) 1eabcd

180 cancelcond |= jnp.logical_and(not imag, q % 2 == 0) 1eabcd

181 power = jnp.where(cancelcond, pdiff, x ** -s1) 1eabcd

182 hz = power + hzs # = ζ(s1,x) -/+ ζ(s1,1-x) 1eabcd

183 

184 out = (pi * gam * pha) * hz 1eabcd

185 if imag: 1eabcd

186 out *= jnp.where(neg, -1, 1) 1abcd

187 return out 1eabcd

198def zeta_series_power_diff(x, q, a): 1feabcd

199 """ 

200 compute x^q-a + (-1)^q * [q-th term of 2 * hurwitz_zeta_series(-q, a, x)] 

201 """ 

202 pint = x ** q 1eabcd

203 pz = jnp.where(q, 0, jnp.where(a, -1, 0)) # * 0^q = 0^q-a - 0^q 1eabcd

204 pdif = jnp.where(x, jnp.expm1(-a * jnp.log(x)), pz) # * x^q = x^q-a - x^q 1eabcd

205 gamincr = jnp.where(q, _gamma.gamma_incr(1 + q, -a), 0) 1eabcd

206 # gamincr = Γ(1+q-a) / Γ(1+q)Γ(1-a) - 1 

207 zz = zeta_zero(a) # = ζ(a) - ζ(0) 1eabcd

208 qdif = 2 * (1 + gamincr) * zz - gamincr # = (q-th term) - (q-th term)|_a=0 1eabcd

209 return pint * (pdif + qdif) 1eabcd

211def zeta_zero(s): 1feabcd

212 """ 

213 Compute zeta(s) - zeta(0) for |s| < 1 accurately 

214 """ 

215 

216 # f(s) = zeta(s) - 1 / (s - 1) 

217 # I have the Taylor series of f(s) 

218 # zeta(s) - zeta(0) = f(s) + 1 / (s - 1) + 1/2 = 

219 # = f(s) + 1/(s-1) + 1 - 1 + 1/2 = 

220 # = f(s) - 1/2 + s/(s-1) 

221 

222 t = _patch_jax.float_type(s) 1eabcd

223 coef = jnp.array(_zeta_zero_coef, t).at[0].set(0) 1eabcd

224 fact = jnp.cumprod(jnp.arange(coef.size).at[0].set(1), dtype=t) 1eabcd

225 coef /= fact 1eabcd

226 f = jnp.polyval(coef[::-1], s) 1eabcd

227 return f + s / (s - 1) 1eabcd

249def _gen_zeta_zero_coef(n): # pragma: no cover 1feabcd

250 """ 

251 Compute first n derivatives of zeta(s) - 1/(s-1) at s = 0 

252 """ 

253 import mpmath as mp 

254 with mp.workdps(32): 

255 func = lambda s: mp.zeta(s) - 1 / (s - 1) 

256 return [float(mp.diff(func, 0, k)) for k in range(n)]

270def _zeta_left(s, n): 1feabcd

271 # reflection formula https://dlmf.nist.gov/25.4.E1 

272 m = 1 - n 1eabcd

273 x = -s 1eabcd

274 # m + x = 1 - (n + s) = 1 - n - s 

275 mx = m + x # > 0 1eabcd

276 logpi = -mx * jnp.log(2 * jnp.pi) 1eabcd

277 cos = _cos_pi2(m, x) # = cos(π/2 (m + x)) but accurate for small x 1eabcd

278 loggam = jspecial.gammaln(mx) 1eabcd

279 zet = _zeta_right(x, m) 1eabcd

280 

281 # cancel zeta pole at 1 

282 cos = jnp.where(mx == 1, -jnp.pi / 2, cos) 1eabcd

283 zet = jnp.where(mx == 1, 1, zet) 1eabcd

284 

285 return 2 * jnp.exp(logpi + loggam) * cos * zet 1eabcd
