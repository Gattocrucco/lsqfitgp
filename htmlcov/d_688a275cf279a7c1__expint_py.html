31@functools.partial(jax.custom_jvp, nondiff_argnums=(0,)) 

32def expn_imag(n, x): 

33 """ 

34 Compute E_n(-ix), n integer >= 2, x real >= 0 

35 """ 

36 

37 # expn_imag_smallx loses accuracy due to cancellation between two terms 

38 # ~ x^n-2, while the result ~ x^-1, thus the relative error ~ x^-1/x^n-2 = 

39 # = x^-(n-1) 

40 # 

41 # error of expn_imag_smallx: eps z^n-1 E_1(z) / Gamma(n) ~ 

42 # ~ eps z^n-2 / Gamma(n) 

43 # 

44 # error of expn_asymp: e^-z/z (n)_nt e^z/z^nt-1 E_n+nt(z) = 

45 # = (n)_nt / z^nt E_n+nt(z) ~ 

46 # ~ (n)_nt / z^nt+1 

47 # 

48 # set the errors equal: 

49 # eps z^n-2 / Gamma(n) = (n)_nt / z^nt+1 --> 

50 # --> z = (Gamma(n + nt) / eps)^1/(n+nt-1) 

51 

52 # TODO improve accuracy at large n, it is probably sufficient to use 

53 # something like softmin(1/(n-1), 1/x) e^-ix, where the softmin scale 

54 # increases with n (how?) 

55 

56 x = jnp.asarray(x) 

57 with jax.ensure_compile_time_eval(): 

58 n = jnp.asarray(n) 

59 dt = _patch_jax.float_type(n, x) 

60 if dt == jnp.float32: 

61 nt = jnp.array(10, 'i4') # TODO optimize to raise maximum n 

62 else: 

63 nt = 20 # TODO optimize to raise maximum n 

64 eps = jnp.finfo(dt).eps 

65 knee = (special.gamma(n + nt) / eps) ** (1 / (n + nt - 1)) 

66 small = expn_imag_smallx(n, x) 

67 large = expn_asymp(n, -1j * x, nt) 

68 return jnp.where(x < knee, small, large)

72 

73 # DLMF 8.19.13 

74 

75 x, = primals 

76 xt, = tangents 

77 return expn_imag(n, x), xt * 1j * expn_imag(n - 1, x)

79def expn_imag_smallx(n, x): 

80 

81 # DLMF 8.19.7 

82 

83 n, x = jnp.asarray(n), jnp.asarray(x) 

84 k = jnp.arange(n) 

85 fact = jnp.cumprod(k.at[0].set(1), dtype=_patch_jax.float_type(n, x)) 

86 n_1fact = fact[-1] 

87 ix = 1j * x 

88 E_1 = exp1_imag(x) # E_1(-ix) 

89 E_1 = jnp.where(x, E_1, 0) # Re E_1(-ix) ~ log(x) for x -> 0 

90 part1 = ix ** (n - 1) * E_1 

91 coefs = fact[:-1][(...,) + (None,) * ix.ndim] 

92 part2 = jnp.exp(ix) * jnp.polyval(coefs, ix) 

93 return (part1 + part2) / n_1fact

98def expn_asymp_coefgen(s, e, n): 

99 k = jnp.arange(s, e, dtype=n.dtype) 

100 return (-1) ** k * _gamma.poch(n, k)

102def expn_asymp(n, z, nt): 

103 """ 

104 Compute E_n(z) for large |z|, |arg z| < 3/2 π. ``nt`` is the number of terms 

105 used in the asymptotic series. 

106 """ 

107 

108 # DLMF 8.20.2 

109 

110 invz = 1 / z 

111 return jnp.exp(-z) * invz * _taylor.taylor(expn_asymp_coefgen, (n,), 0, nt, invz)

209def _si_smallx(x): 

210 """ Compute Si(x) = int_0^x dt sin t / t, for x < 4""" 

211 x2 = jnp.square(x) 

212 dtype = _patch_jax.float_type(x) 

213 num = jnp.polyval(jnp.array(_si_num[::-1], dtype), x2) 

214 denom = jnp.polyval(jnp.array(_si_denom[::-1], dtype), x2) 

215 return x * num / denom

217def _minus_cin_smallx(x): 

218 """ Compute -Cin(x) = int_0^x dt (cos t - 1) / t, for x < 4 """ 

219 x2 = jnp.square(x) 

220 dtype = _patch_jax.float_type(x) 

221 num = jnp.polyval(jnp.array(_ci_num[::-1], dtype), x2) 

222 denom = jnp.polyval(jnp.array(_ci_denom[::-1], dtype), x2) 

223 return x2 * num / denom

225def _ci_smallx(x): 

226 """ Compute Ci(x) = -int_x^oo dt cos t / t, for x < 4 """ 

227 gamma = 0.57721566490153286060 

228 return gamma + jnp.log(x) + _minus_cin_smallx(x)

230def _f_largex(x): 

231 """ Compute f(x) = int_0^oo dt sin t / (x + t), for x > 4 """ 

232 x2 = 1 / jnp.square(x) 

233 dtype = _patch_jax.float_type(x) 

234 num = jnp.polyval(jnp.array(_f_num[::-1], dtype), x2) 

235 denom = jnp.polyval(jnp.array(_f_denom[::-1], dtype), x2) 

236 return num / denom / x

238def _g_largex(x): 

239 """ Compute g(x) = int_0^oo dt cos t / (x + t), for x > 4 """ 

240 x2 = 1 / jnp.square(x) 

241 dtype = _patch_jax.float_type(x) 

242 num = jnp.polyval(jnp.array(_g_num[::-1], dtype), x2) 

243 denom = jnp.polyval(jnp.array(_g_denom[::-1], dtype), x2) 

244 return x2 * num / denom

246def _exp1_imag_smallx(x): 

247 """ Compute E_1(-ix), for x < 4 """ 

248 return -_ci_smallx(x) + 1j * (jnp.pi / 2 - _si_smallx(x))

250def _exp1_imag_largex(x): 

251 """ Compute E_1(-ix), for x > 4 """ 

252 s = jnp.sin(x) 

253 c = jnp.cos(x) 

254 f = _f_largex(x) 

255 g = _g_largex(x) 

256 real = -f * s + g * c 

257 imag = f * c + g * s 

258 return real + 1j * imag # e^ix (g + if)

260@jax.jit 

261def exp1_imag(x): 

262 """ 

263 Compute E_1(-ix) = int_1^oo dt e^ixt / t, for x > 0 

264 Reference: Rowe et al. (2015, app. B) 

265 """ 

266 return jnp.where(x < 4, _exp1_imag_smallx(x), _exp1_imag_largex(x))
