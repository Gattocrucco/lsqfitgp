35 x_train, 

36 y_train, 

37 *, 

38 x_test=None, 

39 fitkw={}, 

40 kernelkw={}): 

41 """ 

42 GP version of BART. 

43 

44 Evaluate a Gaussian process regression with a kernel which accurately 

45 approximates the infinite trees limit of BART. The hyperparameters are 

46 optimized to their marginal MAP. 

47 

48 Parameters 

49 ---------- 

50 x_train : (n, p) or dataframe 

51 Observed covariates. 

52 y_train : (n,) 

53 Observed outcomes. 

54 x_test : (n*, p) or dataframe, optional 

55 Covariates of outcomes to be imputed. 

56 fitkw : dict 

57 Additional arguments passed to `empbayes_fit`, overrides the 

58 defaults. 

59 kernelkw : dict 

60 Additional arguments passed to `BART`, overrides the 

61 defaults. 

62  

63 Attributes 

64 ---------- 

65 yhat_train_mean : (n,) array 

66 The posterior mean of the latent regression function at the observed 

67 covariates. 

68 yhat_train_var : (n,) array 

69 The posterior variance of the latent regression function at the 

70 observed covariates. 

71 yhat_test_mean : (n*,) array 

72 The posterior mean of the latent regression function at the 

73 covariates of imputed outcomes. 

74 yhat_test_var : (n*,) array 

75 The posterior variance of the latent regression function at the 

76 covariates of imputed outcomes. 

77 sigma : gvar 

78 The error term standard deviation marginal MAP. 

79 alpha : gvar 

80 The numerator of the tree spawn probability (named ``base`` in 

81 BayesTree and BART). 

82 beta : gvar 

83 The depth exponent of the tree spawn probability (named ``power`` in 

84 BayesTree and BART). 

85 meansdev : gvar 

86 The prior standard deviation of the latent regression function. 

87 fit : empbayes_fit 

88 The hyperparameters fit object. 

89 gp : GP 

90 The centered Gaussian process object constructed at the 

91 hyperparameters MAP. Its keys are 'trainmean' and 'testmean'. 

92 mu : scalar 

93 The prior mean. 

94 

95 Notes 

96 ----- 

97 The tree splitting grid is set using quantiles of the observed 

98 covariates. This corresponds to settings ``usequants=True``, 

99 ``numcut=inf`` in the R packages BayesTree and BART. 

100 

101 See also 

102 -------- 

103 BART 

104  

105 """ 

106 # convert covariate matrices to StructuredArray 

107 x_train = self._to_structured(x_train) 

108 if x_test is None: 

109 x_test = _array.StructuredArray(numpy.empty_like(x_train, shape=0)) 

110 else: 

111 x_test = self._to_structured(x_test) 

112 assert x_train.dtype == x_test.dtype 

113 assert x_train.ndim == x_test.ndim == 1 

114 assert x_train.size > len(x_train.dtype) 

115 

116 # make sure data is a 1d array 

117 if hasattr(y_train, 'to_numpy'): 

118 y_train = y_train.to_numpy() 

119 y_train = y_train.squeeze() # for dataframes 

120 y_train = jnp.asarray(y_train) 

121 assert y_train.shape == x_train.shape 

122 

123 # prior mean and variance 

124 ymin = jnp.min(y_train) 

125 ymax = jnp.max(y_train) 

126 mu_mu = (ymax + ymin) / 2 

127 k_sigma_mu = (ymax - ymin) / 2 

128 

129 # splitting points and indices 

130 splits = _kernels.BART.splits_from_coord(x_train) 

131 def toindices(x): 

132 ix = _kernels.BART.indices_from_coord(x, splits) 

133 return _array.unstructured_to_structured(ix, names=x.dtype.names) 

134 i_train = toindices(x_train) 

135 i_test = toindices(x_test) 

136 

137 # prior on hyperparams 

138 hyperprior = { 

139 '__bayestree__B(alpha)': copula.beta('__bayestree__B', 2, 1), 

140 '__bayestree__IG(beta)': copula.invgamma('__bayestree__IG', 1, 1), 

141 'log(k)': gvar.gvar(numpy.log(2), 2), 

142 'log(sigma2)': gvar.gvar(numpy.log(1), 2), 

143 } 

144 

145 # GP factory 

146 def makegp(hp, *, i_train, i_test, splits): 

147 kw = dict(alpha=hp['alpha'], beta=hp['beta'], maxd=10, reset=[2,4,6,8], gamma=0.95) 

148 kw.update(kernelkw) 

149 kernel = _kernels.BART(splits=splits, indices=True, **kw) 

150 kernel *= (k_sigma_mu / hp['k']) ** 2 

151 

152 gp = _GP.GP(kernel, checkpos=False, checksym=False, solver='chol') 

153 gp.addx(i_train, 'trainmean') 

154 gp.addcov(hp['sigma2'] * jnp.eye(i_train.size), 'noise') 

155 gp.addtransf({'trainmean': 1, 'noise': 1}, 'data') 

156 gp.addx(i_test, 'testmean') 

157 

158 return gp 

159 

160 # fit hyperparameters 

161 info = {'data': y_train - mu_mu} 

162 options = dict( 

163 verbosity=3, 

164 raises=False, 

165 jit=True, 

166 minkw=dict(method='l-bfgs-b', options=dict(maxls=4, maxiter=100)), 

167 mlkw=dict(epsrel=0), 

168 forward=True, 

169 ) 

170 options.update(fitkw) 

171 gpkw=dict(i_train=i_train, i_test=i_test, splits=splits) 

172 fit = _fit.empbayes_fit(hyperprior, makegp, info, gpfactorykw=gpkw, **options) 

173 

174 # extract hyperparameters from minimization result 

175 self.sigma = gvar.sqrt(fit.p['sigma2']) 

176 self.alpha = fit.p['alpha'] 

177 self.beta = fit.p['beta'] 

178 self.meansdev = k_sigma_mu / fit.p['k'] 

179 

180 # set attributes 

181 gp = makegp(fit.pmean, **gpkw) 

182 yhat_train_mean, yhat_train_cov = gp.predfromdata(info, 'trainmean', raw=True) 

183 self.yhat_train_mean = mu_mu + yhat_train_mean 

184 self.yhat_train_var = jnp.diag(yhat_train_cov) 

185 yhat_test_mean, yhat_test_cov = gp.predfromdata(info, 'testmean', raw=True) 

186 self.yhat_test_mean = mu_mu + yhat_test_mean 

187 self.yhat_test_var = jnp.diag(yhat_test_cov) 

188 self.fit = fit 

189 self.gp = gp 

190 self.mu = mu_mu

199 return _array.StructuredArray(x)
