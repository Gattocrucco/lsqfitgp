37 weights=None,

40 """ 

41 GP version of BART. 

42 

43 Evaluate a Gaussian process regression with a kernel which accurately 

44 approximates the infinite trees limit of BART. The hyperparameters are 

45 optimized to their marginal MAP. 

46 

47 Parameters 

48 ---------- 

49 x_train : (n, p) array or dataframe 

50 Observed covariates. 

51 y_train : (n,) array 

52 Observed outcomes. 

53 weights : (n,) array 

54 Weights used to rescale the error variance (as 1 / weight). 

55 fitkw : dict 

56 Additional arguments passed to `~lsqfitgp.empbayes_fit`, overrides 

57 the defaults. 

58 kernelkw : dict 

59 Additional arguments passed to `~lsqfitgp.BART`, overrides the 

60 defaults. 

61  

62 Attributes 

63 ---------- 

64 mu : scalar 

65 The prior mean. 

66 sigma : gvar 

67 The error term standard deviation. If there are weights, the sdev 

68 for each unit is obtained dividing ``sigma`` by sqrt(weight). 

69 alpha : gvar 

70 The numerator of the tree spawn probability (named ``base`` in 

71 BayesTree and BART). 

72 beta : gvar 

73 The depth exponent of the tree spawn probability (named ``power`` in 

74 BayesTree and BART). 

75 meansdev : gvar 

76 The prior standard deviation of the latent regression function. 

77 fit : empbayes_fit 

78 The hyperparameters fit object. 

79 

80 Methods 

81 ------- 

82 gp : 

83 Create a GP object. 

84 data : 

85 Creates the dictionary to be passed to `GP.pred` to represent 

86 ``y_train``. 

87 pred : 

88 Evaluate the regression function at given locations. 

89 

90 Notes 

91 ----- 

92 The tree splitting grid is set using quantiles of the observed 

93 covariates. This corresponds to settings ``usequants=True``, 

94 ``numcut=inf`` in the R packages BayesTree and BART. Use the 

95 ``kernelkw`` parameter to customize the grid. 

96 

97 See also 

98 -------- 

99 lsqfitgp.BART 

100  

101 """ 

102 

103 # convert covariates to StructuredArray 

104 x_train = self._to_structured(x_train) 

105 

106 # convert outcomes to 1d array 

107 if hasattr(y_train, 'to_numpy'): 

108 y_train = y_train.to_numpy() 

109 y_train = y_train.squeeze() # for dataframes 

110 y_train = jnp.asarray(y_train) 

111 assert y_train.shape == x_train.shape 

112 

113 # check weights 

114 if weights is None: 

115 weights = jnp.ones_like(y_train) 

116 assert weights.shape == y_train.shape 

117 

118 # prior mean and variance 

119 ymin = jnp.min(y_train) 

120 ymax = jnp.max(y_train) 

121 mu_mu = (ymax + ymin) / 2 

122 k_sigma_mu = (ymax - ymin) / 2 

123 

124 # splitting points and indices 

125 splits = _kernels.BART.splits_from_coord(x_train) 

126 i_train = self._toindices(x_train, splits) 

127 

128 # prior on hyperparams 

129 sigma2_priormean = numpy.mean((y_train - y_train.mean()) ** 2 * weights) 

130 hyperprior = { 

131 '__bayestree__B(alpha)': copula.beta('__bayestree__B', 2, 1), 

132 # base of tree gen prob 

133 '__bayestree__IG(beta)': copula.invgamma('__bayestree__IG', 1, 1), 

134 # exponent of tree gen prob 

135 'log(k)': gvar.gvar(numpy.log(2), 2), 

136 # denominator of prior sdev 

137 'log(sigma2)': gvar.gvar(numpy.log(sigma2_priormean), 2), 

138 # i.i.d. error variance, scaled with weights 

139 'mu': gvar.gvar(mu_mu, k_sigma_mu), 

140 # mean of the GP 

141 } 

142 

143 # GP factory 

144 def makegp(hp, *, i_train, weights, splits): 

145 kw = dict( 

146 alpha=hp['alpha'], beta=hp['beta'], 

147 maxd=10, reset=[2, 4, 6, 8], gamma=0.95, 

148 ) 

149 kw.update(kernelkw) 

150 kernel = _kernels.BART(splits=splits, indices=True, **kw) 

151 kernel *= (k_sigma_mu / hp['k']) ** 2 

152 

153 gp = _GP.GP(kernel, checkpos=False, checksym=False, solver='chol') 

154 gp.addx(i_train, 'trainmean') 

155 gp.addcov(jnp.diag(hp['sigma2'] / weights), 'trainnoise') 

156 gp.addtransf({'trainmean': 1, 'trainnoise': 1}, 'train') 

157 

158 return gp 

159 

160 # data factory 

161 def info(hp, **_): 

162 return {'train': y_train - hp['mu']} 

163 

164 # fit hyperparameters 

165 gpkw = dict(i_train=i_train, weights=weights, splits=splits) 

166 options = dict( 

167 verbosity=3, 

168 raises=False, 

169 jit=True, 

170 minkw=dict(method='l-bfgs-b', options=dict(maxls=4, maxiter=100)), 

171 mlkw=dict(epsrel=0), 

172 forward=True, 

173 gpfactorykw=gpkw, 

174 ) 

175 options.update(fitkw) 

176 fit = _fit.empbayes_fit(hyperprior, makegp, info, **options) 

177 

178 # extract hyperparameters from minimization result 

179 self.sigma = gvar.sqrt(fit.p['sigma2']) 

180 self.alpha = fit.p['alpha'] 

181 self.beta = fit.p['beta'] 

182 self.meansdev = k_sigma_mu / fit.p['k'] 

183 self.mu = fit.p['mu'] 

184 

185 # set public attributes 

186 self.fit = fit 

187 

188 # set private attributes 

189 self._splits = splits 

190 self._ystd = y_train.std() 

191 

192 def _gethp(self, hp, rng): 1abcdef

193 if not isinstance(hp, str): 

194 return hp 

195 elif hp == 'map': 

196 return self.fit.pmean 

197 elif hp == 'sample': 

198 return _fastraniter.sample(self.fit.pmean, self.fit.pcov, rng=rng) 

199 else: 

200 raise KeyError(hp)

202 def gp(self, *, hp='map', x_test=None, weights=None, rng=None): 1abcdef

203 """ 

204 Create a Gaussian process with the fitted hyperparameters. 

205 

206 Parameters 

207 ---------- 

208 hp : str or dict 

209 The hyperparameters to use. If ``'map'``, use the marginal maximum a 

210 posteriori. If ``'sample'``, sample hyperparameters from the 

211 posterior. If a dict, use the given hyperparameters. 

212 x_test : array or dataframe, optional 

213 Additional covariates for "test points". 

214 weights : array, optional 

215 Weights for the error variance on the test points. 

216 rng : numpy.random.Generator, optional 

217 Random number generator, used if ``hp == 'sample'``. 

218 

219 Return 

220 ------ 

221 gp : GP 

222 A centered Gaussian process object. To add the mean, use the ``mu`` 

223 attribute of the `bart` object. The keys of the GP are '*mean', 

224 '*noise', and '*', where the "*" stands either for 'train' or 

225 'test'. 

226 """ 

227 

228 # determine hyperparameters 

229 hp = self._gethp(hp, rng) 

230 

231 # create GP object 

232 gp = self.fit.gpfactory(hp, **self.fit.gpfactorykw) 

233 

234 # add test points 

235 if x_test is not None: 

236 

237 # convert covariates to indices 

238 x_test = self._to_structured(x_test) 

239 i_test = self._toindices(x_test, self._splits) 

240 assert i_test.dtype == self.fit.gpfactorykw['i_train'].dtype 

241 

242 # check weights 

243 if weights is not None: 

244 weights = jnp.asarray(weights) 

245 assert weights.shape == i_test.shape 

246 else: 

247 weights = jnp.ones(i_test.shape) 

248 

249 # add test points 

250 gp.addx(i_test, 'testmean') 

251 gp.addcov(jnp.diag(hp['sigma2'] / weights), 'testnoise') 

252 gp.addtransf({'testmean': 1, 'testnoise': 1}, 'test') 

253 

254 return gp

279 def pred(self, *, hp='map', error=False, format='matrices', x_test=None, 1abcdef

280 weights=None, rng=None): 

281 """ 

282 Predict the outcome at given locations. 

283 

284 Parameters 

285 ---------- 

286 hp : str or dict 

287 The hyperparameters to use. If ``'map'``, use the marginal maximum a 

288 posteriori. If ``'sample'``, sample hyperparameters from the 

289 posterior. If a dict, use the given hyperparameters. 

290 error : bool 

291 If ``False`` (default), make a prediction for the latent mean. If 

292 ``True``, add the error term.  

293 format : {'matrices', 'gvar'} 

294 If 'matrices' (default), return the mean and covariance matrix 

295 separately. If 'gvar', return an array of `GVar`s. 

296 x_test : array or dataframe, optional 

297 Covariates for the locations where the prediction is computed. If 

298 not specified, predict at the data covariates. 

299 weights : array, optional 

300 Weights for the error variance on the test points. 

301 rng : numpy.random.Generator, optional 

302 Random number generator, used if ``hp == 'sample'``. 

303 

304 Return 

305 ------ 

306 If ``format`` is 'matrices' (default): 

307 

308 mean, cov : arrays 

309 The mean and covariance matrix of the Normal posterior distribution 

310 over the regression function at the specified locations. 

311 

312 If ``format`` is 'gvar': 

313 

314 out : array of `GVar` 

315 The same distribution represented as an array of `GVar` objects. 

316 """ 

317 

318 hp = self._gethp(hp, rng) 

319 gp = self.gp(hp=hp, x_test=x_test, weights=weights) 

320 data = self.data(hp=hp) 

321 

322 if x_test is None: 

323 label = 'train' 

324 else: 

325 label = 'test' 

326 if not error: 

327 label += 'mean' 

328 

329 if format == 'gvar': 

330 out = gp.predfromdata(data, label, keepcorr=False) 

331 return out + hp['mu'] 

332 elif format == 'matrices': 

333 outmean, outcov = gp.predfromdata(data, label, raw=True) 

334 return outmean + hp['mu'], outcov 

335 else: 

336 raise KeyError(format)

341 # convert to StructuredArray 

342 if hasattr(x, 'columns'): 

343 x = _array.StructuredArray.from_dataframe(x) 

344 elif x.dtype.names is None: 

345 x = _array.unstructured_to_structured(x) 

346 else:

349 # check 

350 assert x.ndim == 1

352 def check_numerical(path, dtype): 

353 if not numpy.issubdtype(dtype, numpy.number): 

354 raise TypeError(f'covariate `{path}` is not numerical') 

355 cls._walk_dtype(x.dtype, check_numerical) 

356 

357 return x

373 def __repr__(self): 1abcdef

374 weights = numpy.array(self.fit.gpfactorykw['weights']) 

375 avgsigma = numpy.sqrt(numpy.mean(self.sigma ** 2 / weights)) 

376 return f"""BART fit: 

377alpha = {self.alpha} (0 -> intercept only, 1 -> any) 

378beta = {self.beta} (0 -> any, ∞ -> no interactions) 

379mean = {self.mu} 

380latent sdev = {self.meansdev} (large -> conservative extrapolation) 

381data total sdev = {self._ystd:.3g} 

382error sdev (avg weighted) = {avgsigma} 

383error sdev (unweighted) = {self.sigma}"""
