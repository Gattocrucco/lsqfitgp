32def raniter(mean, cov, n=None, eps=None): 

33 """ 

34  

35 Take random samples from a multivariate gaussian. 

36  

37 This generator mimics the interface of `gvar.raniter`, but takes as input 

38 the mean and covariance separately instead of a collection of gvars. 

39  

40 Parameters 

41 ---------- 

42 mean : scalar, array, or dictionary of scalars/arrays 

43 The mean of the gaussian distribution. 

44 cov : scalar, array, or dictionary of scalars/arrays 

45 The covariance matrix. If `mean` is a dictionary, `cov` must be a 

46 dictionary with pair of keys from `mean` as keys. 

47 n : int, optional 

48 The maximum number of iterations. Default unlimited. 

49 eps : float, optional 

50 Used to correct the eigenvalues of the covariance matrix to handle 

51 non-positivity due to roundoff, relative to the largest eigenvalue. 

52 Default is number of variables times floating point epsilon. 

53  

54 Yields 

55 ------ 

56 samp : scalar, array, or dictionary of scalars/arrays 

57 The random sample in the same format of `mean`. 

58  

59 Examples 

60 -------- 

61  

62 >>> mean = {'a': np.arange(3)} 

63 >>> cov = {('a', 'a'): np.eye(3)} 

64 >>> for sample in lgp.raniter(mean, cov, 3): 

65 >>> print(sample) 

66  

67 """ 

68 

69 # convert mean and cov to 1d and 2d arrays 

70 if hasattr(mean, 'keys'): # a dict or gvar.BufferDict 

71 if not hasattr(mean, 'buf'): 71 ↛ 73line 71 didn't jump to line 73, because the condition on line 71 was never false

72 mean = gvar.BufferDict(mean) 

73 flatmean = mean.buf 

74 squarecov = np.empty((len(flatmean), len(flatmean))) 

75 for k1 in mean: 

76 slic1 = mean.slice(k1) 

77 for k2 in mean: 

78 slic2 = mean.slice(k2) 

79 squarecov[slic1, slic2] = cov[k1, k2] 

80 else: # an array or scalar 

81 mean = np.array(mean, copy=False) 

82 cov = np.array(cov, copy=False) 

83 flatmean = mean.reshape(-1) 

84 squarecov = cov.reshape(len(flatmean), len(flatmean)) 

85 

86 # decompose the covariance matrix 

87 try: 

88 covdec = _linalg.CholGersh(squarecov, eps=eps) 

89 except np.linalg.LinAlgError: 

90 warnings.warn('covariance matrix not positive definite with eps={}'.format(eps)) 

91 covdec = _linalg.EigCutFullRank(squarecov, eps=eps) 

92 

93 # take samples 

94 iterable = itertools.count() if n is None else range(n) 

95 for _ in iterable: 

96 iidsamp = np.random.randn(len(flatmean)) 

97 samp = flatmean + covdec.correlate(iidsamp) 

98 

99 # pack the samples with the original shape 

100 if hasattr(mean, 'keys'): 

101 samp = gvar.BufferDict(mean, buf=samp) 

102 else: 

103 samp = samp.reshape(mean.shape) if mean.shape else samp.item 

104 

105 yield samp
