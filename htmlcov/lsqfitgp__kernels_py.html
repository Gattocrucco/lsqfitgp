178@isotropickernel(input='soft', derivable=_matern_derivable) 

179def Matern(r, nu=None): 

180 """ 

181 Matérn kernel of real order.  

182  

183 .. math:: 

184 k(r) = \\frac {2^{1-\\nu}} {\\Gamma(\\nu)} x^\\nu K_\\nu(x), 

185 \\quad \\nu = \\texttt{nu} > 0, 

186 \\quad x = \\sqrt{2\\nu} r 

187  

188 The nearest integer below `nu` indicates how many times the gaussian 

189 process is derivable: so for `nu` < 1 it is continuous but not derivable, 

190 for 1 <= `nu` < 2 it is derivable but has not a decond derivative, etc. The 

191 half-integer case (nu = 1/2, 3/2, ...) uses internally a simpler formula so 

192 you should prefer it. Also, taking derivatives of the process is supported 

193 only for half-integer nu. 

194 

195 """ 

196 assert np.isscalar(nu) 

197 assert nu > 0 

198 x = np.sqrt(2 * nu) * r 

199 if (2 * nu) % 1 == 0: 

200 return _maternp(x, int(nu - 1/2)) 

201 else: 

202 return 2 ** (1 - nu) / special.gamma(nu) * x ** nu * special.kv(nu, x)

347@kernel(forcekron=True) 

348def Gibbs(x, y, scalefun=lambda x: 1): 

349 """ 

350 Gibbs kernel. 

351  

352 .. math:: 

353 k(x, y) = \\sqrt{ \\frac {2 s(x) s(y)} {s(x)^2 + s(y)^2} } 

354 \\exp \\left( -\\frac {(x - y)^2} {s(x)^2 + s(y)^2} \\right), 

355 \\quad s = \\texttt{scalefun}. 

356  

357 Kernel which in some sense is like a gaussian kernel where the scale 

358 changes at every point. The scale is computed by the parameter `scalefun` 

359 which must be a callable taking the x array and returning a scale for each 

360 point. By default `scalefun` returns a constant so it is a gaussian kernel. 

361  

362 """ 

363 sx = scalefun(x) 

364 sy = scalefun(y) 

365 assert np.all(sx > 0) 

366 assert np.all(sy > 0) 

367 denom = sx ** 2 + sy ** 2 

368 factor = np.sqrt(2 * sx * sy / denom) 

369 return factor * np.exp(-(x - y) ** 2 / denom)

602  

603 .. math:: 

604 k(x, y) &= \\frac1{\\zeta(2n)} \\sum_{k=1}^\\infty

610 &= (-1)^{n+1}

612 B_{2n}(x-y \\mod 1), 

613  

614 where :math:`B_s(x)` is a Bernoulli polynomial. It is equivalent to fitting 

615 with a Fourier series of period 1 with independent priors on the 

616 coefficients `k` with mean zero and standard deviation :math:`1/k^n`. The 

617 process is :math:`n - 1` times derivable. 

618  

619 Note that the :math:`k = 0` term is not included in the summation, so the 

620 mean of the process over one period is forced to be zero. 

621  

622 """

644@kernel(forcekron=True) 

645def OrnsteinUhlenbeck(x, y): 

646 """ 

647 Ornstein-Uhlenbeck process kernel. 

648  

649 .. math:: 

650 k(x, y) = \\exp(-|x - y|) - \\exp(-(x + y)), 

651 \\quad x, y \\ge 0 

652

671@kernel(forcekron=True, derivable=_Celerite_derivable) 

672def Celerite(x, y, gamma=1, B=0): 

673 """ 

674 Celerite kernel. 

675  

676 .. math:: 

677 k(x, y) = \\exp(-\\gamma|x - y|) 

678 \\big( \\cos(x - y) + B \\sin(|x - y|) \\big) 

679  

680 This is the covariance function of an AR(2) process with complex roots. The 

681 parameters must satisfy the condition :math:`|B| \\le \\gamma`. For 

682 :math:`B = \\gamma` it is equivalent to the :class:`Harmonic` kernel with 

683 :math:`\\eta Q = 1/B, Q > 1`, and it is derivable. 

684  

685 Reference: Daniel Foreman-Mackey, Eric Agol, Sivaram Ambikasaran, and Ruth 

686 Angus: *Fast and Scalable Gaussian Process Modeling With Applications To 

687 Astronomical Time Series*. 

688 """ 

689 

690 assert np.isscalar(gamma) and 0 <= gamma < np.inf 

691 assert np.isscalar(B) and np.isfinite(B) 

692 assert np.abs(B) <= gamma 

693 tau = _Kernel._abs(x - y) 

694 return np.exp(-gamma * tau) * (np.cos(tau) + B * np.sin(tau))

717def Harmonic(x, y, Q=1): 

718 """ 

719 Damped stochastically driven harmonic oscillator kernel. 

720  

721 .. math::

725 \\cosh(\\eta\\tau) + \\sinh(\\eta\\tau) / (\\eta Q) 

726 & 0 < Q < 1 \\\\ 

727 1 + \\tau & Q = 1 \\\\ 

728 \\cos(\\eta\\tau) + \\sin(\\eta\\tau) / (\\eta Q) 

729 & Q > 1, 

730 \\end{cases} 

731  

732 where :math:`\\tau = |x - y|` and :math:`\\eta = \\sqrt{|1 - 1/Q^2|}`. 

733  

734 The process is the solution to the stochastic differential equation 

735  

736 .. math:: f''(x) + 2/Q f'(x) + f(x) = w(x), 

737  

738 where `w` is white noise. 

739  

740 The parameter `Q` is the quality factor, i.e. the ratio between the energy 

741 stored in the oscillator and the energy lost in each cycle due to damping. 

742 The angular frequency is 1, i.e. the period is 2π. The process is derivable 

743 one time. 

744  

745 In 1D, for `Q` = 1 (default) and `scale` = sqrt(1/3), it is the Matérn 3/2 

746 kernel. 

747

749 Angus: *Fast and Scalable Gaussian Process Modeling With Applications To 

750 Astronomical Time Series*.
