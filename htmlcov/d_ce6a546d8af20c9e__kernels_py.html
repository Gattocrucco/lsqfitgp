157@isotropickernel(input='soft', derivable=_matern_derivable) 

158def Matern(r, nu=None): 

159 """ 

160 Matérn kernel of real order.  

161  

162 .. math:: 

163 k(r) = \\frac {2^{1-\\nu}} {\\Gamma(\\nu)} x^\\nu K_\\nu(x), 

164 \\quad \\nu = \\texttt{nu} > 0, 

165 \\quad x = \\sqrt{2\\nu} r 

166  

167 The nearest integer below `nu` indicates how many times the Gaussian 

168 process is derivable: so for `nu` < 1 it is continuous but not derivable, 

169 for 1 <= `nu` < 2 it is derivable but has not a decond derivative, etc. The 

170 half-integer case (nu = 1/2, 3/2, ...) uses internally a simpler formula so 

171 you should prefer it. Also, taking derivatives of the process is supported 

172 only for half-integer nu. 

173 

174 """ 

175 assert np.isscalar(nu) 

176 assert nu > 0 

177 x = np.sqrt(2 * nu) * r 

178 if (2 * nu) % 1 == 0: 

179 return _maternp(x, int(nu - 1/2)) 

180 else: 

181 return 2 ** (1 - nu) / special.gamma(nu) * x ** nu * special.kv(nu, x)

275 """ 

276 Neural network kernel. 

277  

278 .. math:: 

279 k(x, y) = \\frac 2 \\pi 

280 \\arcsin \\left( \\frac 

281 {

283 }{

285 (1 + 2 (q + y \\cdot y))

326@kernel(forcekron=True) 

327def Gibbs(x, y, scalefun=lambda x: 1): 

328 """ 

329 Gibbs kernel. 

330  

331 .. math:: 

332 k(x, y) = \\sqrt{ \\frac {2 s(x) s(y)} {s(x)^2 + s(y)^2} } 

333 \\exp \\left( -\\frac {(x - y)^2} {s(x)^2 + s(y)^2} \\right), 

334 \\quad s = \\texttt{scalefun}. 

335  

336 Kernel which in some sense is like a Gaussian kernel where the scale 

337 changes at every point. The scale is computed by the parameter `scalefun` 

338 which must be a callable taking the x array and returning a scale for each 

339 point. By default `scalefun` returns a constant so it is a Gaussian kernel. 

340  

341 """ 

342 sx = scalefun(x) 

343 sy = scalefun(y) 

344 assert np.all(sx > 0) 

345 assert np.all(sy > 0) 

346 denom = sx ** 2 + sy ** 2 

347 factor = np.sqrt(2 * sx * sy / denom) 

348 return factor * np.exp(-(x - y) ** 2 / denom)

551@autograd.extend.primitive

555 binom = special.binom(n, k)

577@kernel(forcekron=True, derivable=_fourier_derivable) 

578def Fourier(x, y, n=2): 

579 """ 

580 Fourier kernel. 

581  

582 .. math:: 

583 k(x, y) &= \\frac1{\\zeta(2n)} \\sum_{k=1}^\\infty 

584 \\frac {\\cos(2\\pi kx)}{k^n} \\frac {\\cos(2\\pi ky)}{k^n} 

585 + \\frac1{\\zeta(2n)} \\sum_{k=1}^\\infty 

586 \\frac {\\sin(2\\pi kx)}{k^n} \\frac {\\sin(2\\pi ky)}{k^n} = \\\\ 

587 &= \\frac1{\\zeta(2n)} \\sum_{k=1}^\\infty 

588 \\frac {\\cos(2\\pi k(x-y))} {k^{2n}} = \\\\ 

589 &= (-1)^{n+1} 

590 \\frac1{\\zeta(2n)} \\frac {(2\\pi)^{2n}} {2(2n)!} 

591 B_{2n}(x-y \\mod 1), 

592  

593 where :math:`B_s(x)` is a Bernoulli polynomial. It is equivalent to fitting 

594 with a Fourier series of period 1 with independent priors on the 

595 coefficients `k` with mean zero and standard deviation :math:`1/k^n`. The 

596 process is :math:`n - 1` times derivable. 

597  

598 Note that the :math:`k = 0` term is not included in the summation, so the 

599 mean of the process over one period is forced to be zero. 

600  

601 """ 

602 

603 # TODO maxk parameter to truncate the series. I have to manually sum the 

604 # components. 

605 

606 # TODO parity = {None, 'even', 'odd'} to keep only sines/cosines. I would 

607 # keep the normalization to variance = 1. Maybe this is general and should 

608 # be implemented for any kernel as a kernel function, like `where`. 

609 

610 # TODO this appears to be less numerically accurate than other kernels. 

611 # It's probably the Bernoulli polynomial computation, find out if it is due 

612 # to computing the polynomial using the coefficients or if the coefficients 

613 # given by scipy are inaccurate. 

614 

615 assert isinstance(n, (int, np.integer)) 

616 assert n >= 1 

617 s = 2 * n 

618 diff = (x - y) % 1 

619 sign0 = -(-1) ** n 

620 factor = (2 * np.pi) ** s / (2 * special.factorial(s) * special.zeta(s)) 

621 return sign0 * factor * _bernoulli_poly(s, diff)

623@kernel(forcekron=True)

625 """

642def _Celerite_derivable(**kw): 

643 gamma = kw.get('gamma', 1) 

644 B = kw.get('B', 0) 

645 if np.isscalar(gamma) and np.isscalar(B) and B == gamma: 

646 return 1 

647 else: 

648 return False

650@kernel(forcekron=True, derivable=_Celerite_derivable) 

651def Celerite(x, y, gamma=1, B=0): 

652 """ 

653 Celerite kernel. 

654  

655 .. math:: 

656 k(x, y) = \\exp(-\\gamma|x - y|) 

657 \\big( \\cos(x - y) + B \\sin(|x - y|) \\big) 

658  

659 This is the covariance function of an AR(2) process with complex roots. The 

660 parameters must satisfy the condition :math:`|B| \\le \\gamma`. For 

661 :math:`B = \\gamma` it is equivalent to the :class:`Harmonic` kernel with 

662 :math:`\\eta Q = 1/B, Q > 1`, and it is derivable. 

663  

664 Reference: Daniel Foreman-Mackey, Eric Agol, Sivaram Ambikasaran, and Ruth 

665 Angus: *Fast and Scalable Gaussian Process Modeling With Applications To 

666 Astronomical Time Series*. 

667 """ 

668 

669 assert np.isscalar(gamma) and 0 <= gamma < np.inf 

670 assert np.isscalar(B) and np.isfinite(B) 

671 assert np.abs(B) <= gamma 

672 tau = _Kernel._abs(x - y) 

673 return np.exp(-gamma * tau) * (np.cos(tau) + B * np.sin(tau))

695@kernel(forcekron=True, derivable=1) 

696def Harmonic(x, y, Q=1): 

697 """ 

698 Damped stochastically driven harmonic oscillator kernel. 

699  

700 .. math:: 

701 k(x, y) = 

702 \\exp\\left( -\\frac {\\tau} {Q} \\right) 

703 \\begin{cases} 

704 \\cosh(\\eta\\tau) + \\sinh(\\eta\\tau) / (\\eta Q) 

705 & 0 < Q < 1 \\\\ 

706 1 + \\tau & Q = 1 \\\\ 

707 \\cos(\\eta\\tau) + \\sin(\\eta\\tau) / (\\eta Q) 

708 & Q > 1, 

709 \\end{cases} 

710  

711 where :math:`\\tau = |x - y|` and :math:`\\eta = \\sqrt{|1 - 1/Q^2|}`. 

712  

713 The process is the solution to the stochastic differential equation 

714  

715 .. math:: f''(x) + 2/Q f'(x) + f(x) = w(x), 

716  

717 where `w` is white noise. 

718  

719 The parameter `Q` is the quality factor, i.e., the ratio between the energy 

720 stored in the oscillator and the energy lost in each cycle due to damping. 

721 The angular frequency is 1, i.e., the period is 2π. The process is derivable 

722 one time. 

723  

724 In 1D, for `Q` = 1 (default) and `scale` = sqrt(1/3), it is the Matérn 3/2 

725 kernel. 

726  

727 Reference: Daniel Foreman-Mackey, Eric Agol, Sivaram Ambikasaran, and Ruth 

728 Angus: *Fast and Scalable Gaussian Process Modeling With Applications To 

729 Astronomical Time Series*. 

730 """

736 

737 assert np.isscalar(Q) and 0 < Q < np.inf 

738 

739 tau = _Kernel._abs(x - y) 

740 

741 if Q < 1/2: 

742 etaQ = np.sqrt((1 - Q) * (1 + Q)) 

743 tauQ = tau / Q 

744 pexp = np.exp(_sqrt1pm1(-np.square(Q)) * tauQ)
